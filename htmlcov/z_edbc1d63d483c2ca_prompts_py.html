8_CLI_SUFFIX = ( 

9 "\nDo NOT write any scripts, use any tools, or read any files." 

10 "\nPrint your output directly to stdout." 

11) 

12 

13 

14@dataclass(frozen=True) 

15class PromptTemplate: 

16 """Backend-aware prompt template. 

17 

18 ``body`` is shared across both backends. 

19 ``cli_suffix`` is appended only in CLI mode (tool/script constraints). 

20 """ 

21 

22 body: str 

23 cli_suffix: str = _CLI_SUFFIX 

24 

25 

26class PromptBackend(enum.Enum): 

27 CLI = "cli" 

28 API = "api" 

29 

30 

31def render_prompt(template: PromptTemplate, backend: PromptBackend, **kwargs: str) -> str: 

32 """Render *template* for the given *backend*, substituting *kwargs* placeholders.""" 

33 suffix = template.cli_suffix if backend == PromptBackend.CLI else "" 

34 return (template.body + suffix).format(**kwargs)

93RECAP_MAP_PROMPT = PromptTemplate( 

94 body="""\ 

95You are a senior news editor. Group ALL headlines below into blocks. 

96 

97Group headlines so that the essential information from all articles in a block \

99what happened from the title alone. 

100 

101Headlines about the same event, the same country's affairs, or the same ongoing \ 

102story belong in one block if they can be jointly described in 1-3 informative \ 

103sentences. 

104 

105Do NOT mix genuinely unrelated events. "Swiss resort fire and Venice carnival" \ 

106in one block is unacceptable. 

107Do NOT create newspaper-section blocks like "Sports and Culture" or "World News". \ 

108Each block must describe specific news, not a topic category. 

109Do NOT mix genuinely unrelated news in one block. 

110 

111GOOD title: "Heavy snow hits Serbia; most roads cleared but the Valjevo road \ 

112remains blocked." 

113BAD title: "Flooding in Nepal" — no details on casualties or scale. 

114BAD title: "10,000 programmers laid off" — where? why? 

115BAD: "Serbian news" — not informative. 

116BAD: "Global politics" — a category, not an event description. 

117BAD: "Crime and Legal Matters" — this is a newspaper section, not a story. 

118BAD: "Travel, Culture and Sports" — unrelated topics dumped together. 

119BAD: "Diverse Global Developments" — meaningless catch-all. 

120If a headline does not fit any group, it is better as a single-article block \ 

121with an informative title than forced into an unrelated category. 

122 

123Every headline number must appear in exactly one block. 

124Do not skip any headline. 

125 

126FOLLOW (give extra attention): {follow_policy}

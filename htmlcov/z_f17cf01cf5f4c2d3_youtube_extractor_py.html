45@dataclass(slots=True) 

46class TranscriptResult: 

47 """Result of YouTube transcript extraction.""" 

48 

49 text: str 

50 language: str 

51 is_success: bool

54 @property 

55 def is_blocked(self) -> bool: 

56 """True when YouTube blocked the request (IP ban / bot detection).""" 

57 return self.error == IP_BLOCKED_ERROR

60def _blocked(video_id: str) -> TranscriptResult: 

61 logger.warning("YouTube blocked request for %s (IP ban / bot detection)", video_id) 

62 return TranscriptResult(text="", language="", is_success=False, error=IP_BLOCKED_ERROR)

65def _permanent_failure(video_id: str, error_code: str) -> TranscriptResult: 

66 """Return a definitive failure for a known, unactionable transcript error.""" 

67 logger.debug("No transcript for %s: %s", video_id, error_code) 

68 return TranscriptResult(text="", language="", is_success=False, error=error_code)

71def _classify_permanent(exc: Exception) -> str | None: 

72 """Return an error code if *exc* is a known permanent failure, else ``None``.""" 

73 for exc_type, code in _PERMANENT_ERRORS.items(): 

74 if isinstance(exc, exc_type): 

75 return code 

76 return None

79def extract_video_id(url: str) -> str | None: 

80 """Extract YouTube video ID from URL, or None if not a YouTube URL. 

81 

82 >>> extract_video_id("https://www.youtube.com/watch?v=dQw4w9WgXcQ") 

83 'dQw4w9WgXcQ' 

84 >>> extract_video_id("https://www.youtube.com/shorts/iz4C9oE0wTk") 

85 'iz4C9oE0wTk' 

86 >>> extract_video_id("https://example.com") is None 

87 True 

88 """ 

89 match = _YT_VIDEO_ID_RE.search(url) 

90 return match.group(1) if match else None

93def is_youtube_url(url: str) -> bool: 

94 """Check if URL points to a YouTube video (including Shorts).""" 

95 return extract_video_id(url) is not None

99 """Check if URL points to a YouTube Shorts video (no transcripts)."""

103def _fetch_preferred( 

104 api: YouTubeTranscriptApi, 

105 video_id: str, 

106 languages: tuple[str, ...], 

107 max_chars: int, 

108) -> TranscriptResult | None: 

109 """Try the high-level ``fetch()`` with preferred languages. 

110 

111 Returns ``None`` when the preferred languages are not available 

112 (so the caller can fall back), or a definitive result otherwise. 

113 """ 

114 try: 

115 fetched = api.fetch(video_id, languages=list(languages)) 

116 text = " ".join(snippet.text for snippet in fetched) 

117 lang = fetched.language_code if hasattr(fetched, "language_code") else languages[0] 

118 if max_chars > 0 and len(text) > max_chars: 

119 text = text[:max_chars].rstrip() 

120 return TranscriptResult(text=text, language=lang, is_success=True) 

121 except RequestBlocked: 

122 return _blocked(video_id) 

123 except Exception as exc: # noqa: BLE001 

124 code = _classify_permanent(exc) 

125 if code is not None: 

126 return _permanent_failure(video_id, code) 

127 logger.debug("Primary transcript fetch failed for %s, trying fallback", video_id) 

128 return None

131def _fetch_any( 

132 api: YouTubeTranscriptApi, 

133 video_id: str, 

134 max_chars: int, 

135) -> TranscriptResult: 

136 """Try every available transcript variant as a fallback.""" 

137 try: 

138 transcript_list = api.list(video_id) 

139 except RequestBlocked: 

140 return _blocked(video_id) 

141 except Exception as exc: # noqa: BLE001 

142 code = _classify_permanent(exc) 

143 if code is not None: 

144 return _permanent_failure(video_id, code) 

145 logger.warning("Transcript retrieval failed for %s: %s", video_id, type(exc).__name__) 

146 return TranscriptResult( 

147 text="", 

148 language="", 

149 is_success=False, 

150 error=TRANSCRIPT_RETRIEVAL_FAILED_ERROR, 

151 ) 

152 

153 for transcript in transcript_list: 

154 try: 

155 fetched = transcript.fetch() 

156 text = " ".join(snippet.text for snippet in fetched) 

157 if max_chars > 0 and len(text) > max_chars: 

158 text = text[:max_chars].rstrip() 

159 return TranscriptResult(text=text, language=transcript.language_code, is_success=True) 

160 except RequestBlocked: 

161 return _blocked(video_id) 

162 except Exception: # noqa: BLE001 

163 logger.debug("Transcript variant %s failed for %s", transcript.language_code, video_id) 

164 

165 return TranscriptResult( 

166 text="", 

167 language="", 

168 is_success=False, 

169 error=NO_TRANSCRIPTS_ERROR, 

170 )

173def fetch_transcript( 

174 url: str, 

175 *, 

176 languages: tuple[str, ...] = PREFERRED_LANGUAGES, 

177 max_chars: int = 0, 

178) -> TranscriptResult: 

179 """Fetch transcript/subtitles for a YouTube video URL. 

180 

181 Uses youtube-transcript-api v1.x: instance-based API with dataclass snippets. 

182 Tries preferred languages first via the high-level ``fetch()`` method, 

183 then falls back to the first available transcript. 

184 

185 On ``RequestBlocked`` / ``IpBlocked`` the result carries 

186 ``error=IP_BLOCKED_ERROR`` so the caller can abort early. 

187 """ 

188 video_id = extract_video_id(url) 

189 if not video_id: 

190 return TranscriptResult( 

191 text="", 

192 language="", 

193 is_success=False, 

194 error=f"not a YouTube URL: {url}", 

195 ) 

196 

197 if is_shorts_url(url): 

198 return TranscriptResult( 

199 text="", 

200 language="", 

201 is_success=False, 

202 error="YouTube Shorts do not have transcripts", 

203 ) 

204 

205 api = YouTubeTranscriptApi() 

206 result = _fetch_preferred(api, video_id, languages, max_chars) 

207 if result is not None: 

208 return result 

209 return _fetch_any(api, video_id, max_chars)
